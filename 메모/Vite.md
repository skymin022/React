# vite 설치 
npm install -g create-vite

# vite 프로젝트 생성
npm create vite

# vite 프로젝트 실행 
cd vite-project
npm install
npm run dev

# React 프로젝트 생성 
# create-vite 프로젝트 명 --template react ## 이걸로 ㄱㄱ
create-vite vite-react-project --template react

# 프로젝트 이동 
cd vite-react-project 

# 의존성 설치
npm install

# 서버 실행 
npm run dev
